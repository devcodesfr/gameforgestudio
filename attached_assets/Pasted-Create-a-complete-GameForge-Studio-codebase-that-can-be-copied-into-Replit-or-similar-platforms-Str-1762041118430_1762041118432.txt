// Navigation Functions
function setupNavigation() {
    const navItems = document.querySelectorAll('.nav-item');
    
    navItems.forEach(item => {
        item.addEventListener('click', function(e) {
            e.preventDefault();
            
            const section = this.getAttribute('data-section');
            if (section && section !== currentSection) {
                switchToSection(section);
            }
        });
    });
}

function switchToSection(sectionName) {
    console.log(`Switching to section: ${sectionName}`);
    
    // Hide all content sections
    const sections = document.querySelectorAll('.content-section');
    sections.forEach(section => {
        section.classList.remove('active');
    });
    
    // Remove active state from all nav items
    const navItems = document.querySelectorAll('.nav-item');
    navItems.forEach(item => {
        item.classList.remove('active');
    });
    
    // Show target section
    const targetSection = document.getElementById(sectionName);
    if (targetSection) {
        targetSection.classList.add('active');
        currentSection = sectionName;
        
        // Add active state to clicked nav item
        const activeNavItem = document.querySelector(`[data-section="${sectionName}"]`);
        if (activeNavItem) {
            activeNavItem.classList.add('active');
        }
        
        console.log(`Successfully switched to ${sectionName}`);
    } else {
        console.error(`Section ${sectionName} not found`);
    }
}

// Project Interaction Functions
function setupProjectInteractions() {
    // Use event delegation for project tiles
    document.addEventListener('click', function(e) {
        const projectTile = e.target.closest('.project-tile');
        const actionBtn = e.target.closest('.btn-action');
        
        // Handle project tile clicks (but not action buttons)
        if (projectTile && !actionBtn) {
            const projectId = projectTile.getAttribute('data-project-id');
            if (projectId && projects[projectId]) {
                showProjectDetails(projectId);
            }
        }
        
        // Handle action button clicks
        if (actionBtn) {
            e.stopPropagation();
            const projectTile = actionBtn.closest('.project-tile');
            const projectId = projectTile ? projectTile.getAttribute('data-project-id') : null;
            const action = actionBtn.textContent.toLowerCase().trim();
            
            handleProjectAction(projectId, action);
        }
    });
}

        li.textContent = component;
        componentsList.appendChild(li);
    });
    
    // Update action button
    const actionBtn = document.getElementById('modal-action-btn');

function debounce(func, wait) {
    let timeout;
    return function executedFunction(...args) {
        const later = () => {
            clearTimeout(timeout);
            func(...args);
        };
        clearTimeout(timeout);
        timeout = setTimeout(later, wait);
    };
}

    // Debounce function for search input
    debounce(func, wait) {
        let timeout;
        return function executedFunction(...args) {
            const later = () => {
                clearTimeout(timeout);
                func(...args);
            };
            clearTimeout(timeout);
            timeout = setTimeout(later, wait);
        };
    },

    // Generate random ID
    generateId() {
        return 'id_' + Math.random().toString(36).substr(2, 9);
    },

    // Format date
    formatDate(date) {
        const now = new Date();
        const diff = now - date;
        const days = Math.floor(diff / (1000 * 60 * 60 * 24));
        const hours = Math.floor(diff / (1000 * 60 * 60));
        const minutes = Math.floor(diff / (1000 * 60));

        if (days > 0) return `${days} day${days > 1 ? 's' : ''} ago`;
        if (hours > 0) return `${hours} hour${hours > 1 ? 's' : ''} ago`;
        if (minutes > 0) return `${minutes} minute${minutes > 1 ? 's' : ''} ago`;
        return 'Just now';
    },

    // Sanitize HTML
    sanitizeHtml(text) {
        const div = document.createElement('div');

    }

// Modal Management
class ModalManager {
    static showModal(modalId) {
        const modal = document.getElementById(modalId);
        if (modal) {
            modal.classList.add('show');
            document.body.style.overflow = 'hidden';
            
            // Focus trap
            this.trapFocus(modal);
        }
    }

    static closeModal(modal) {
        if (modal) {
            modal.classList.remove('show');
            document.body.style.overflow = '';
            
            // Clear form if it's the new project modal
            if (modal.id === 'newProjectModal') {
                const form = modal.querySelector('#newProjectForm');
                if (form) form.reset();
            }
        }
    }

    static trapFocus(modal) {
        const focusableElements = modal.querySelectorAll(
            'button, [href], input, select, textarea, [tabindex]:not([tabindex="-1"])'
        );
        const firstElement = focusableElements[0];
        const lastElement = focusableElements[focusableElements.length - 1];

        const handleTabKey = (event) => {
            if (event.key === 'Tab') {
                if (event.shiftKey) {
                    if (document.activeElement === firstElement) {
                        event.preventDefault();
                        lastElement.focus();
                    }
                } else {
                    if (document.activeElement === lastElement) {
                        event.preventDefault();
                        firstElement.focus();
                    }
                }
            } else if (event.key === 'Escape') {
                this.closeModal(modal);
            }
        };

        modal.addEventListener('keydown', handleTabKey);
        
        // Focus first element
        if (firstElement) {
            setTimeout(() => firstElement.focus(), 100);
        }

        // Remove event listener when modal closes
        const observer = new MutationObserver((mutations) => {
            mutations.forEach((mutation) => {
                if (mutation.attributeName === 'class') {
                    if (!modal.classList.contains('show')) {
                        modal.removeEventListener('keydown', handleTabKey);
                        observer.disconnect();
                    }
                }
            });
        });
        observer.observe(modal, { attributes: true });
    }
}

            const current = Math.floor(start + (end - start) * progress);
            
            element.textContent = current;
            
            if (progress < 1) {
                requestAnimationFrame(animate);
            }
        };

    static createProject(form) {
        try {
            const formData = new FormData(form);
            const project = {
                id: Utils.generateId(),
                title: formData.get('projectName'),
                engine: formData.get('projectEngine'),
                genre: formData.get('projectGenre') || 'other',
                description: formData.get('projectDescription') || 'No description provided.',
                progress: 0,
                lastModified: new Date(),
                downloads: 0,
                likes: 0,
                image: this.getEngineIcon(formData.get('projectEngine'))
            };

            // Validate required fields
            if (!project.title || !project.engine) {
                throw new Error('Please fill in all required fields.');
            }

            // Add to projects
            AppState.projects.myProjects.unshift(project);
            
            // Close modal and refresh view
            ModalManager.closeModal(document.getElementById('newProjectModal'));
            
            // Switch to my projects if not already there
            if (AppState.currentProjectType !== 'myProjects') {
                this.switchProjectType('myProjects');
            } else {
                this.renderProjects();
            }
            
            // Show success message
            EventManager.prototype.showNotification(`Project "${project.title}" created successfully!`, 'success');
            
        } catch (error) {
            console.error('Error creating project:', error);
            EventManager.prototype.showError(error.message || 'Failed to create project. Please try again.');
        }
    }
